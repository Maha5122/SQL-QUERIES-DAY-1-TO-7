CREATE OR REPLACE PROCEDURE EMPLOYEE(FIRST_NAME IN VARCHAR2,SALARY IN NUMBER) AS
BEGIN
FOR I IN (SELECT FIRST_NAME,SALARY FROM HR.EMPLOYEES)
    LOOP    
     IF I.SALARY > 10000 THEN
      DBMS_OUTPUT.PUT_LINE(I.FIRST_NAME||'SALARY IS'||I.SALARY);
     END IF;
    END LOOP;
END;

EXEC EMPLOYEE('Steven',15000);

4. FUNCTION

CREATE OR REPLACE FUNCTION MIN3(N1 IN NUMBER,N2 IN NUMBER,N3 IN NUMBER) RETURN NUMBER IS

BEGIN
 IF N1 < N2 AND N1 < N3 THEN
    RETURN N1;
 ELSIF N2 < N3 THEN
    RETURN N2;
 ELSE
    RETURN N3;
 END IF;
END; 

SELECT MIN3(7,4,9) FROM DUAL;

DECLARE
    A NUMBER := 4;
    B NUMBER := 60;
    C NUMBER := 9;
    D NUMBER := 54;
BEGIN 
    IF A>B AND A>C AND A>D THEN
      DBMS_OUTPUT.PUT_LINE('A IS BIG');
    ELSIF B>C AND B>D THEN
      DBMS_OUTPUT.PUT_LINE('B IS BIG');
    ELSIF C>D THEN
      DBMS_OUTPUT.PUT_LINE('C IS BIG');
    ELSE
      DBMS_OUTPUT.PUT_LINE('D IS GREATER');
    END IF;
END;
